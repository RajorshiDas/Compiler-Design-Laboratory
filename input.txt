@@ Switch case example using decide-when-otherwise
num day = 3
decide (day)
  when 1
    write("Monday")
  when 2
    write("Tuesday")
  when 3
    write("Wednesday")
  when 4
    write("Thursday")
  when 5
    write("Friday")
  otherwise
    write("Weekend")
end
