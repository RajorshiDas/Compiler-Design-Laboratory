@@ Valid test: mixed full program in report syntax.
@@ Expected interpreter output:
@@ square = 1
@@ square = 4
@@ square = 9
@@ finished

#load mathLib.fx
start() -> empty
{
    num i = 1;
    num value;

    fx square(num x) -> num
    {
        send x * x;
    }

    repeat(i, i <= 3, i + 1)
    {
        value = square(i);
        write("square = ", value);
    }

    decide i
    {
        when(4) => write("finished");
        otherwise => write("unexpected");
    }

    send 0;
}
