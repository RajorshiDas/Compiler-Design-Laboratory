// Valid test: decide / when / otherwise.
// Expected AST behavior: decide node with two when clauses and one otherwise clause.
// Expected interpreter output:
// two

num x = 2;

decide (x) {
    when x == 1 then {
        write("one");
    }
    when x == 2 then {
        write("two");
    }
    otherwise then {
        write("other");
    }
}
