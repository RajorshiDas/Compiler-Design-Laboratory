@@ Valid test: report-style decide / when / otherwise.
@@ Expected interpreter output:
@@ Option B selected

#load mathLib.fx
start() -> empty
{
    chr option = 'B';

    decide option
    {
        when('A') => write("Option A selected");
        when('B') => write("Option B selected");
        otherwise => write("Other option selected");
    }

    send 0;
}
