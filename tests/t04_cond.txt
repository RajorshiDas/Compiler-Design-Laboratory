@@ Valid test: report-style chk / else_try / then with logical operators.
@@ Expected interpreter output:
@@ a is greater than b
@@ a equals b or is greater

#load mathLib.fx
start() -> empty
{
    num a = 10;
    num b = 5;

    chk (a > b AND b < 10)
        write("a is greater than b");
    end chk

    else_try(a < b)
        write("a is less than b");
    end else_try
    then
        write("a equals b or is greater");
    end then

    send 0;
}
