// Valid test: repeat-until loop.
// Expected AST behavior: one repeat statement with two assignments in the body.
// Expected interpreter output:
// 1
// 3
// 6
// Expected IR shape:
// L1:
// i = i + 1
// sum = sum + i
// write sum
// tN = i == 3
// ifFalse tN goto L1

num i = 0;
num sum = 0;

repeat doing {
    i = i + 1;
    sum = sum + i;
    write(sum);
} until (i == 3);
