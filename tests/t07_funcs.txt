@@ Valid test: functions with declared return types and builtin math helpers.
@@ Expected interpreter output:
@@ 15
@@ 100
@@ 4

#load mathLib.fx
start() -> empty;
{
    num a = 10;
    num b = 5;
    num result;

    fx addNumbers(num x, num y) -> num
    {
        send x + y;
    }

    result = addNumbers(a, b);
    write(result);

    result = power(a, 2);
    write(result);

    result = squart(16);
    write(result);
    send 0;
}
