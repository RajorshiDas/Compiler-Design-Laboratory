@@ Valid test: arithmetic, modulus, XOR, and absolute value.
@@ Expected interpreter output:
@@ 11
@@ 2
@@ 7
@@ 5

#load mathLib.fx
start() -> empty
{
    num a = 5;
    num b = 2;
    num c;

    c = a + b * 3;
    write(c);

    c = 17 % 5;
    write(c);

    c = a XOR b;
    write(c);

    c = |-5|;
    write(c);
    send 0;
}
