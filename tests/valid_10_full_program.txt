// Valid test: mixed full program.
// Expected AST behavior: function definition, repeat-until, chk/else_try, and decide in one program.
// Expected interpreter output:
// small
// 1
// big
// 4
// big
// 9
// finished

fx square(num x) {
    return x * x;
}

num i = 0;
num value;

repeat doing {
    i = i + 1;
    value = square(i);

    chk (value > 2) then {
        write("big");
    } else_try {
        write("small");
    }

    write(value);
} until (i == 3);

decide (i) {
    when i == 3 then {
        write("finished");
    }
    otherwise then {
        write("unexpected");
    }
}
