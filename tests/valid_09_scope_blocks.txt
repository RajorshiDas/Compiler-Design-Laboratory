@@ Valid test: nested blocks and scope shadowing in report-style program.
@@ Expected interpreter output:
@@ 10

@@ 7

@@ 10

#load mathLib.fx
start() -> empty
{
    num x = 10;

    write(x);

    {
        num x = 99;
        num y = x + 1;
        write(y);

        {
            num x = 7;
            write(x);
        }

        write(x);
    }

    write(x);
    send 0;
}
